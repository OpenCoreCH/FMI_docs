170 template <typename T>
171 raw_func convert_to_raw_function(FMI::Utils::Function<T> f, std::size_t size_in_bytes) {
172 auto func = [f](char* a, char* b) -> void {
173 T* dest = reinterpret_cast<T*>(a);
174 *dest = f(*((T*) a), *((T*) b));
175 };
176 return func;
177 }

180 template <typename A>
181 raw_func convert_to_raw_function(FMI::Utils::Function<std::vector<A>> f, std::size_t size_in_bytes) {
182 auto func = [f, size_in_bytes](char* a, char* b) -> void {
183 std::vector<A> vec_a((A*) a, (A*) (a + size_in_bytes));
184 std::vector<A> vec_b((A*) b, (A*) (b + size_in_bytes));
185 std::vector<A> res = f(vec_a, vec_b);
186 std::memcpy(a, (char*) res.data(), size_in_bytes);
187 };
188 return func;
189 }
